/* Inline member functions.
What is the programs output */


#include <iostream>
#include <string>
using namespace std;

class City {
   public:
      void SetName(string cityName) {
         name = cityName;
      }
      void SetPopulation(int cityPopulation) {
         population = cityPopulation;
      }
      void Print() const;

   private:
      string name;
      int population;
};

void City::Print() const {
   cout << name << ". Population: " << population << endl;
}

int main() {
   City cityToVisit;

   cityToVisit.SetName("Lusaka");
   cityToVisit.SetPopulation(1331260);

   cityToVisit.Print();

   return 0;
}

// Lusaka. Population: 1331260
