   cin >> length1;
   cin >> length2;

   /* Your code goes here */
   if (length1 < length2) {
      lengthPointer = &length1;
   } else if (length2 < length1) {
      lengthPointer = &length2;
   }

   if (lengthPointer == nullptr) {
      cout << "The lengths are the same." << endl;
   }
   else {
      cout << fixed << setprecision(1) << *lengthPointer << " is the smaller length." << endl;
      
      if (lengthPointer == &length1) {
         cout << "lengthPointer points to length1." << endl;
      }
      if (lengthPointer == &length2) {

      }
   }

   return 0;
}

#include <iostream>

void ChangeLevel(char* levelPointer, char levelValue) {

   /* Your code goes here */
   if (levelPointer == nullptr) {
      cout << "levelPointer is null." << endl;
   } else {
      *levelPointer = levelValue;
   }

}

int main() {

      levelPointer = &level;
